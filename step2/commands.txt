docker build -t my-php-fpm php/

docker run -d \
  --name script \
  --network tp3-net \
  -v $(pwd)/app:/app \
  my-php-fpm

docker run -d \
  --name data \
  --network tp3-net \
  -e MARIADB_ROOT_PASSWORD=password \
  -v $(pwd)/db:/docker-entrypoint-initdb.d \
  mariadb:latest

docker run -d \
  --name http \
  --network tp3-net \
  -p 8080:8080 \
  -v $(pwd)/app:/app \
  -v $(pwd)/../step1/nginx/default.conf:/etc/nginx/conf.d/default.conf \
  nginx:alpine
